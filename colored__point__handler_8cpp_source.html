73void ColoredPointHandler::UpdatePointColor()
74{
75 switch (coloring_mode_)
76 {
77 case DEFAULT:
78 {
79 current_point_radius_ = default_point_radius_;
80 point_color_ = Scalar(0, 0, 255);
81 break;
82 }
83 case COLORING_BASED_ON_FREQUENCY:
84 {
85 current_point_radius_ = default_point_radius_;
86 // переведём диапазон частоты [0; range_limit] в [0; 255] int color value;
87 if (!frequencies_.empty())
88 {
89 point_color_ = HelperFunctions::RatioToRgb(main_frequency_ / (sampling_rate_ / 2));
90 }
91 break;
92 }
93 case COLORING_BASED_ON_AMPLITUDE:
94 {
95 current_point_radius_ = default_point_radius_;
96 // переведём диапазон амплитуд [0; max_amplitude_] в [0; 255] int color value;
97 double current_amplitude_absolute = sqrt(current_amplitude_.x * current_amplitude_.x + current_amplitude_.y * current_amplitude_.y);
98 double ratio = (current_amplitude_absolute - extremum_amplitude_.second) / (extremum_amplitude_.first - extremum_amplitude_.second);
99 point_color_ = HelperFunctions::RatioToRgb(ratio);
100 if (point_color_ == Scalar(0, 0, 0))
101 {
102 std::cout << "ratio: " << ratio << std::endl;
103 std::cout << "current_amplitude_.x: " << current_amplitude_.x << std::endl;
104 std::cout << "max ampl: " << extremum_amplitude_.first << std::endl;
105 std::cout << "min ampl: " << extremum_amplitude_.second << std::endl;
106 }
107 break;
108 }
109 }
110 if (interacted_)
111 {
112 point_color_ = Scalar(255, 255, 255);
113 current_point_radius_ *= 3;
114 }
115}
