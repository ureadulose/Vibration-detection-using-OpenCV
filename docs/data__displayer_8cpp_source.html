15void DataDisplayer::DrawData(Mat& frame)
16{
17 if (previous_points_coordinates_.size() != vec_point_handlers_.size())
18 {
19 std::cout << "DEBUG: ERROR - previous_points_coordinates_ size doesnt match vec_point_handlers_ size" << std::endl;
20 return;
21 }
22 double res_mp_ = 1.5;
23 for (int i = 0; i < vec_point_handlers_.size(); i++)
24 {
25 Scalar circle_color;
26 Point2f new_point = vec_point_handlers_[i].GetLastFoundCoordinates();
27 Point2f text_coordinates = vec_point_handlers_[i].GetTextCoordinates();
28 Point3f amplitude = vec_point_handlers_[i].GetCurrentAmplitude();
29 if (vec_point_handlers_[i].IsInteracted(last_mouse_position_))
30 {
31 circle_color = Scalar(255, 150, 50);
32 }
33 else circle_color = Scalar(255, 75, 25);
34
35 // отрисовываем линии точек
36 line(frame, previous_points_coordinates_[i], new_point, Scalar(0, 255, 0), 1, LINE_AA);
37
38 // отрисовываем круг вокруг точки
39 circle(frame, new_point, 10, circle_color, 2);
40
41 // отрисовываем частоту вибрации
42 putText(
43 frame,
44 "hz: " + std::to_string(vec_point_handlers_[i].GetCurrentVibrationFrequency()),
45 Point(text_coordinates.x + 15, text_coordinates.y + res_mp_ * i * 20 + 25),
46 FONT_HERSHEY_PLAIN,
47 res_mp_ * 1.25,
48 Scalar(0, 69, 255),
49 2
50 );
51
52 // отрисовываем амплитуды вибрации
53 putText(
54 frame,
55 "x: " + std::to_string(amplitude.x),
56 Point(text_coordinates.x, text_coordinates.y - res_mp_ * 20),
57 FONT_HERSHEY_PLAIN,
58 res_mp_ * 1,
59 Scalar(0, 255, 255),
60 2
61 );
62 putText(
63 frame,
64 "y: " + std::to_string(amplitude.y),
65 Point(text_coordinates.x, text_coordinates.y - res_mp_ * 2 * 20),
66 FONT_HERSHEY_PLAIN,
67 res_mp_ * 1,
68 Scalar(0, 255, 255),
69 2
70 );
71 putText(
72 frame,
73 "z: " + std::to_string(amplitude.z),
74 Point(text_coordinates.x, text_coordinates.y - res_mp_ * 3 * 20),
75 FONT_HERSHEY_PLAIN,
76 res_mp_ * 1,
77 Scalar(0, 255, 255),
78 2
79 );
80 }
81}

246std::vector<int> DataDisplayer::Rgb(double ratio)
247{
248 // we want to normalize ratio so that it fits in to 6 regions
249 // where each region is 256 units long
250 int normalized = int(ratio * 256 * 4);
251
252 // find the region for this position
253 int region = normalized / 256;
254
255 // find the distance to the start of the closest region
256 int x = normalized % 256;
257
258 int r = 0, g = 0, b = 0;
259 switch (region)
260 {
261 case 0: r = 255; g = 0; b = 0; g += x; break;
262 case 1: r = 255; g = 255; b = 0; r -= x; break;
263 case 2: r = 0; g = 255; b = 0; b += x; break;
264 case 3: r = 0; g = 255; b = 255; g -= x; break;
265 //case 4: r = 0; g = 0; b = 255; r += x; break;
266 //case 5: r = 255; g = 0; b = 255; b -= x; break;
267 }
268 return { r, g, b };
269}
