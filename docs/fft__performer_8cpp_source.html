30std::vector<float> FftPerformer::ExecuteFft(int sampling_frequency, bool absolute_peak, int debugging_num)
31{
32 //std::cout << "doin " << debugging_num << std::endl;
33
34 std::vector<Point2f> fft_result;
35 std::vector<Point2f> p2;
36 std::vector<Point2f> p1;
37 std::vector<float> frequencies;
38 std::vector<int> indexes_of_peak_frequencies;
39 std::vector<float> magnitudes;

41 std::vector<float> peak_frequencies;
42
43 this->sampling_frequency_ = sampling_frequency;
44 this->size_of_vecs_ = coordinates_of_point_.size();
45
46 // subtracting mean value so the samples would be fluctuating around zero
47 std::vector<cv::Point2f> vec_meaned_coordinates_of_point_;
48
49 mean_coordinates_of_point_ = mean(coordinates_of_point_);
50 //std::cout << mean_coordinates_of_point_ << std::endl;
51 double meaned_x = mean_coordinates_of_point_.val[0];
52 double meaned_y = mean_coordinates_of_point_.val[1];
53 //std::cout << meaned_x << std::endl;
54 //std::cout << meaned_y << std::endl;
55 amplitude_x_ = coordinates_of_point_.back().x - meaned_x;
56 amplitude_y_ = coordinates_of_point_.back().y - meaned_y;
57
58 for (int i = 0; i < coordinates_of_point_.size(); i++)
59 {
60
61 vec_meaned_coordinates_of_point_.push_back(Point2f(coordinates_of_point_[i].x - meaned_x, coordinates_of_point_[i].y - meaned_y));
62 //std::cout << vec_meaned_coordinates_of_point_[i] << std::endl;
63 }
64
65 dft(vec_meaned_coordinates_of_point_, fft_result);
66
67 // computing two-sided spectrum P2
68 for (int i = 0; i < fft_result.size(); i++)
69 {
70 Point2f tmp;
71 tmp.x = abs(fft_result[i].x / size_of_vecs_);
72 tmp.y = abs(fft_result[i].y / size_of_vecs_);
73 p2.push_back(tmp);
74 }
75
76 // computing single-sided spectrum P1 based on P2 and the even-valued signal length (the same as size_of_vecs_)
77 for (int i = 0; i < p2.size(); i++)
78 {
79 size_t idx = 0;
80 idx = (int)(((double)(i)) / 2.0 + 1.0);
81 p1.push_back(p2[idx]);
82 }
83
84 // possible frequencies
85 for (int i = 0; i < size_of_vecs_; i++)
86 {
87 float tmp = sampling_frequency_ * ((double)(i) / 2) / size_of_vecs_;
88 frequencies.push_back(tmp);
89 }
90
91 for (int i = 0; i < p1.size(); i++)
92 {
93 float current_magnitude = sqrt(p1[i].x * p1[i].x + p1[i].y * p1[i].y);
94 magnitudes.push_back(current_magnitude);
95 }
96
98 std::cout << "coords of point size is: " << coordinates_of_point_.size() << std::endl;
99 if (coordinates_of_point_.size() > 200)
100 {
101 std::fstream file;
102 file.open("C:/Users/seeyo/source/repos/Visual-Vibration-Tracking-V2/docs/magnitudes.txt", 'w');
103
104 for (int i = 0; i < magnitudes.size(); i++)
105 {
106 std::string tmp = std::to_string(frequencies[i]) + " " + std::to_string(magnitudes[i]) + " " + std::to_string(p1[i].x) + " " + std::to_string(p1[i].y);
107 file << tmp << std::endl;
108 }
109 std::cout << "written written written written written written written written" << std::endl;
110 file.close();
111
112 WriteFftDataToTxt();
113 }
114
116
117 if (!absolute_peak)
118 {
119 // searching peaks in output vector of magnitudes
120 PeakFinder::findPeaks(magnitudes, indexes_of_peak_frequencies, false, 1);
121
122 // this function works strange, it doesn't always returns the indexes
123 // so i decided to do the following
124 int maxIdx = 0;
125 if (indexes_of_peak_frequencies.empty())
126 {
127 //std::cout << "alternative way" << std::endl;
128 double tmp_max_value = 0.0;
129 // here we start from 1 and not 0
130 // it's called "lyutiy kostyl"
131 for (int i = 1; i < magnitudes.size(); i++)
132 {
133 if (magnitudes[i] > tmp_max_value)
134 {
135 tmp_max_value = magnitudes[i];
136 maxIdx = i;
137 }
138
139 }
140 peak_frequencies.push_back(frequencies[maxIdx]);
141 }
142
143
144 // filling in vector of peak_frequencies with the just found peak frequencies
145 for (int i = 0; i < indexes_of_peak_frequencies.size(); i++)
146 {
147 peak_frequencies.push_back(frequencies[indexes_of_peak_frequencies[i]]);
148
149 //std::cout << "Max is " << frequencies[indexes_of_peak_frequencies[i]] << std::endl;
150 }
151 }
152
153 if (absolute_peak)
154 {
155 int maxIdx = 0;
156 double tmp_max_value = 0.0;
157 // here we start from 1 and not 0
158 // it's called "lyutiy kostyl"
159 for (int i = 1; i < magnitudes.size(); i++)
160 {
161 if (magnitudes[i] > tmp_max_value)
162 {
163 tmp_max_value = magnitudes[i];
164 maxIdx = i;
165 }
166
167 }
168 peak_frequencies.push_back(frequencies[maxIdx]);
169 }
170
171
172
173 return peak_frequencies;
174}
