26cv::Scalar HelperFunctions::RatioToRgb(double ratio)
27{
28 //
29 int normalized = static_cast<int>(ratio * 256 * 4);
30
31 // find the region for this position
32 int region = normalized / 256;
33
34 // find the distance to the start of the closest region
35 int x = normalized % 256;
36
37 int r = 0, g = 0, b = 0;
38
39 switch (region)
40 {
41 case 0: r = 0; g = 0; b = 255; g += x; break; // синий:сине-зеленый
42 case 1: r = 0; g = 255; b = 255; b -= x; break; // сине-зеленый:зеленый
43 case 2: r = 0; g = 255; b = 0; r += x; break; // зеленый:желтый
44 case 3: r = 255; g = 255; b = 0; g -= x; break; // желтый:красный
45 }
46
47 return cv::Scalar(b, g, r);
48}
