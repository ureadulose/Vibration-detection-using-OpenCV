33 vec_meaned_coordinates_of_point_.push_back(Point2f(point_coordinates_[i].x - meaned_x, point_coordinates_[i].y - meaned_y));
34 x_coordinates_.push_back(point_coordinates_[i].x - meaned_x);
35 y_coordinates_.push_back(point_coordinates_[i].y - meaned_y);
36 }
37
38 dft(vec_meaned_coordinates_of_point_, fft_result);
39
40 // computing two-sided spectrum P2
41 for (int i = 0; i < fft_result.size(); i++)
42 {
43 Point2f tmp;
44 tmp.x = abs(fft_result[i].x / samples_amount);
45 tmp.y = abs(fft_result[i].y / samples_amount);
46 p2.push_back(tmp);
47 }
48
49 // computing single-sided spectrum P1 based on P2 and the even-valued signal length (the same as size_of_vecs_)
50 for (int i = 0; i < p2.size(); i++)
51 {
52 size_t idx = 0;
53 idx = (int)(((double)(i)) / 2.0 + 1.0);
54 p1.push_back(p2[idx]);
55 }
56
57 // possible frequencies
58 for (int i = 0; i < samples_amount; i++)
59 {
60 float tmp = sampling_rate_ * ((double)(i) / 2) / samples_amount;
61 frequencies.push_back(tmp);
62 }
63
64 for (int i = 0; i < p1.size(); i++)
65 {
66 float current_magnitude = sqrt(p1[i].x * p1[i].x + p1[i].y * p1[i].y);
67 magnitudes.push_back(current_magnitude);
68 }
69
70 HelperFunctions::DeadzoneFilter(frequencies);
71 HelperFunctions::DeadzoneFilter(magnitudes);
72
74
75 if (point_coordinates_.size() > 5)
76 {
77 std::fstream file;

104 int zero_flag = 0;
105
106 if (!absolute_peak)

108 // searching peaks in output vector of magnitudes
109
110 PeakFinder::findPeaks(magnitudes, indexes_of_peak_frequencies, false, 1);
111
112 int maxIdx = 0;
113

115
116 double mag_sum = 0;
117
118 // filling in vector of peak_frequencies with the just found peak frequencies
119 for (int i = 0; i < indexes_of_peak_frequencies.size(); i++)
120 {
121 peak_frequencies.push_back(frequencies[indexes_of_peak_frequencies[i]]);
122
123 //std::cout << "mag: " << magnitudes[indexes_of_peak_frequencies[i]] << std::endl;
124 mag_sum += magnitudes[indexes_of_peak_frequencies[i]];
125 if ((magnitudes[indexes_of_peak_frequencies[i]]) < 0.01f)
126 zero_flag--;
127 }
128
129 main_frequency_ = frequencies[HelperFunctions::FindGlobalMaxIdx(magnitudes)];
130 float max_diff = HelperFunctions::CalculateMaxDifferenceInVector(magnitudes);
131 mag_max_differences_.push_back(max_diff);
132 float mean_diff_of_max_diffs = 0;
133 if (mag_max_differences_.size() > 1)
134 mean_diff_of_max_diffs = HelperFunctions::CalculateMeanDifferenceInVector(mag_max_differences_);
135
136 if (mean_diff_of_max_diffs < sensivity_ || std::isnan(mean_diff_of_max_diffs))
137 confidence_level_ -= 0.01;
138 else
139 confidence_level_ += 0.01;
140
141 if (zero_flag == 0 && indexes_of_peak_frequencies.size() != 0)
142 {
143 frequencies_.clear();
144 frequencies_.push_back(0.0);
145 }
146 else
147 {
148 frequencies_ = peak_frequencies;
149 }
