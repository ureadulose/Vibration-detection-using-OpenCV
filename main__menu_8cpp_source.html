162void MainMenu::DetectEvent(int event, int x, int y, int flags)
163{
164 // Загружаем названия директорий
165 LoadPaths();
166
167 switch (event)
168 {
169 case EVENT_LBUTTONDOWN:
170 {
171 for (int i = 0; i < buttons_.size(); i++)
172 {
173 // Если клик пользователя был по кнопке
174 if (buttons_[i].contains(Point(x, y)))
175 {
176 switch (i)
177 {
178 case CALIBRATION:
179 {
180 std::cout << "u've chosen camera calibration" << std::endl;
181 ExecuteCameraCalibration(chessboards_path_);
182 break;
183 }
184 case UNDISTORTION:
185 {
186 std::cout << "u've chosen undistortion" << std::endl;
187 ExecuteVideoUndistortion(input_distorted_video_, output_undistorted_video_);
188 break;
189 }
190 case VIBRATION_MONITORING:
191 {
192 std::cout << "u've chosen vibration monitoring" << std::endl;
193 ExecuteVibrationMonitoring(input_vibration_detection_, output_vibration_detection_);
194 break;
195 }
196 case MOVEMENT_ENLARGEMENT:
197 {
198 std::cout << "u've chosen movement enlarging" << std::endl;
199 ExecuteMotionMagnification(input_movement_enlargment_, output_movement_enlargement_);
200 break;
201 }
202 }
203 }
204 }
205 break;
206 }
207 case EVENT_MOUSEMOVE:
208 {
209 last_mouse_position_.x = x;
210 last_mouse_position_.y = y;
211 break;
212 }
213 }
214}

216void MainMenu::Idle()
217{
218 // Устанавливаем хэндлер окна
219 setMouseCallback(winname_, OnMouse, (void*)this);
220
221 while (!mode_chosen_)
222 {

225 if (buttons_[i].contains(last_mouse_position_))
226 {
227 buttons_color_[i] = Scalar(0, 0, 255);
228 }
229 else buttons_color_[i] = Scalar(25, 25, 25);
230 }
231 // Обновляем текущие цвета кнопок
232 UpdateBtnColorAndText();
233
234 imshow(winname_, menu_);
235 switch (waitKey(10))
236 {
237 case 'q':
238 {
239 mode_chosen_ = true;
240 break;
241 }
242 }
243 }
