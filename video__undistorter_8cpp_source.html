158std::vector<std::string> VideoUndistorter::SplitString(std::string input_string, std::string delimiter)
159{
160 int start_position = 0;
161 int end_position;
162 int delimiter_length = delimiter.length();
163
164 std::string token;
165 std::vector<std::string> left;
166
167 while ((end_position = input_string.find(delimiter, start_position)) != std::string::npos) {
168 token = input_string.substr(start_position, end_position - start_position);
169 start_position = end_position + delimiter_length;
170 left.push_back(token);
171 }
172
173 left.push_back(input_string.substr(start_position));
174 return left;
175}
